>>> # import your class
>>> MyList = __import__('1-my_list').MyList

>>> # original 3‐elem list
>>> ml = MyList([3, 1, 2])
>>> ml.print_sorted()
[1, 2, 3]
>>> ml
[3, 1, 2]

>>> # empty list
>>> MyList([]).print_sorted()
[]

>>> # duplicates
>>> MyList([2, 3, 1, 2]).print_sorted()
[1, 2, 2, 3]

>>> # longer example
>>> MyList([10, 5, 7, 5, 1]).print_sorted()
[1, 5, 5, 7, 10]

>>> # test append() and sorted‐print
>>> ml.append(4)
>>> ml
[3, 1, 2, 4]
>>> ml.print_sorted()
[1, 2, 3, 4]

>>> ml.append(0)
>>> ml.print_sorted()
[0, 1, 2, 3, 4]

>>> # now append a negative and ensure it shows up first
>>> ml.append(-5)
>>> ml.print_sorted()
[-5, 0, 1, 2, 3, 4]
